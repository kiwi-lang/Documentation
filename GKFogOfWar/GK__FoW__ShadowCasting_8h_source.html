226inline
227bool UGKShadowCasting::IsBlockingLight(FIntVector BasePos)
228{
229 SCOPE_CYCLE_COUNTER(STAT_SC_QueryIsBlocked);
230
231 auto Terrain = BasePos;
232 auto Blocking = BasePos;
233 Terrain.Z = uint8(EGK_VisbilityLayers::Terrain);
234 Blocking.Z = uint8(EGK_VisbilityLayers::Blocking);
235
236 check(Buffer.Valid(Terrain));
237 check(Buffer.Valid(Terrain));
238
239 return Buffer(Terrain) > BasePos.Z || Buffer(Blocking) >= BasePos.Z;
240}

243inline
244void UGKShadowCasting::SetVisible(FIntVector BufferPos, uint8 TeamId)
245{
246 SCOPE_CYCLE_COUNTER(STAT_SC_SetVisible);
247 uint8 ZHeight = BufferPos.Z & GKFOW_HEIGHT_MASK;
248
249 BufferPos.Z = uint8(EGK_VisbilityLayers::Size) + TeamId;
250 check(Buffer.Valid(BufferPos));
251
252#if WITH_FLYINGUNITS
253 uint8 OldHeight = Buffer(BufferPos) & GKFOW_HEIGHT_MASK;
254 ZHeight = FMath::Max(OldHeight, ZHeight);
255 // Mark this height has visible
256 Buffer(BufferPos) = ZHeight | GKFOW_VISIBLE_MASK;
257#else
258 // Mark this height has visible
259 Buffer(BufferPos) |= GKFOW_VISIBLE_MASK;
260#endif
261}

273{
274 SCOPE_CYCLE_COUNTER(STAT_SC_QueryDistance);
