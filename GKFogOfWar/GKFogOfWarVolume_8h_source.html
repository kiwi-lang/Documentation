463 UFUNCTION(BlueprintCallable, Category = FogOfWar)
464 FORCEINLINE bool IsVisible(FGenericTeamId SeerTeam, FVector Loc) const {
465 SCOPE_CYCLE_COUNTER(STAT_IsVisibleQuery);
466
467 if (bCacheSimilarAllyVision) {
468 class AGKFogOfWarTeam* Team = GetFogOfWarTeam(SeerTeam);
469 class AGKFogOfWarTeam* Ally = GetSimilarAlly(Team);
470
471 if (Ally) {
472 return GKGETATTR(Strategy, IsVisible(Ally->TeamId, Loc), false);
473 }
474 return false;
475 }
476 else {
477 return GKGETATTR(Strategy, IsVisible(SeerTeam, Loc), false);
478 }
479 }
480
481 // Is Actor B visible by Actor A ?
482 FORCEINLINE bool IsVisible(AActor const* A, AActor const* B) const {
483 if (B == nullptr) {
484 // B does not exist quit
485 return false;
486 }
487 const IGenericTeamAgentInterface* AgentA = Cast<const IGenericTeamAgentInterface>(A);
488 const IGenericTeamAgentInterface* AgentB = Cast<const IGenericTeamAgentInterface>(B);
489
490 FGenericTeamId TeamA;
491
492 if (AgentA) {
493 TeamA = AgentA->GetGenericTeamId();
494 } else {
495 // "A" does not belong to a Team, let him see everything
496 return true;
497 }
498
499 if (AgentB) {
500 auto TeamB = AgentB->GetGenericTeamId();
501
502 if (TeamA == TeamB) {
503 // Same Team
504 return true;
505 }
506 }
507
508 // Check if B location is inside the "A" team's vision
509 return IsVisible(TeamA, B->GetActorLocation());
510 }

512 FORCEINLINE bool IsVisible(FGenericTeamId SeerTeam, AActor const *Target) const
513 {
514 if (FGenericTeamId::GetTeamIdentifier(Target) == SeerTeam)
515 return true;
516
517 return IsVisible(SeerTeam, Target->GetActorLocation());
518 }
