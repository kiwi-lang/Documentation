27#define GK_ABILITY_LEVEL_ACCESSOR(Type, Name, Default)\
28 Type Get##Name(int Level) const { return GetAbilityAttributeValue<Type>(Level, this->Name, Type(Default)); }

31template<typename T>
32T GetAbilityAttributeValue(int AbilityLevel, TArray<T> const& Array, T Default = T()) {
33 int Index = FMath::Min(Array.Num(), AbilityLevel) - 1;
34
35 if (Index >= 0) {
36 return Array[Index];
37 }
38 return Default;
39}
