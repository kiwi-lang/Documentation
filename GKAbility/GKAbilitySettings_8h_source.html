52template <typename T>
53void ReadDataTable(class UDataTable *Table, TArray<T *> &Rows)
54{
55 if (Table == nullptr)
56 {
57 return;
58 }
59
60 Rows.Reserve(Table->GetRowMap().Num());
61
62 for (auto RowMapIter(Table->GetRowMap().CreateConstIterator()); RowMapIter; ++RowMapIter)
63 {
64 T *Info = reinterpret_cast<T *>(RowMapIter.Value());
65 Rows.Add(Info);
66 }
67}

69template <typename T>
70T *GetRow(class UDataTable *Table, FName Row)
71{
72 static FString Context;
73 return Table->FindRow<T>(Row, Context, false);
74}
